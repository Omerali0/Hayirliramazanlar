    <script>
      const menuBtn = document.getElementById("menuBtn");
      const sideMenu = document.getElementById("sideMenu");
      const overlay = document.getElementById("overlay");

      menuBtn.addEventListener("click", () => {
        sideMenu.classList.toggle("active");
        overlay.classList.toggle("active");
      });

      overlay.addEventListener("click", () => {
        sideMenu.classList.remove("active");
        overlay.classList.remove("active");
      });

      const ekranGenisligi = window.innerWidth;
      let resimler = [];

      if (ekranGenisligi <= 768) {
        resimler = ["images/dik1.png", "images/dik2.png", "images/dik3.png"];
      } else {
        resimler = ["images/akl.webp", "images/gyt.webp", "images/kkt.webp"];
      }

      resimler.forEach((resim) => {
        const img = new Image();
        img.src = resim;
      });

      let suankiSira = 0;
      let isAnimating = false;

      const mainContentDiv = document.getElementById("mainContent");

      function resimDegistir(yon) {
        if (isAnimating) return;
        isAnimating = true;

        mainContentDiv.classList.add("hide-content");

        setTimeout(() => {
          suankiSira += yon;

          if (suankiSira >= resimler.length) suankiSira = 0;
          if (suankiSira < 0) suankiSira = resimler.length - 1;

          mainContentDiv.style.backgroundImage = `url("${resimler[suankiSira]}")`;

          mainContentDiv.classList.remove("hide-content");

          isAnimating = false;
        }, 400);
      }

      mainContentDiv.style.backgroundImage = `url("${resimler[0]}")`;
      let slaytZamanlayici;

      function donguyuKur(sure) {
        clearInterval(slaytZamanlayici);

        slaytZamanlayici = setInterval(() => {
          resimDegistir(1);
        }, sure);
      }

      donguyuKur(2000);

      mainContentDiv.addEventListener("mouseenter", () => {
        donguyuKur(3000);
      });

      mainContentDiv.addEventListener("mouseleave", () => {
        donguyuKur(2000);
      });
    </script>
